#Fraser Gold Rush
country_event = {
	id = 144300
	trigger = {
		owns = 14 # Vancouver Island
		year = 1853
	}

	fire_only_once = yes

	mean_time_to_happen = {
		months = 60
	}

	title = "EVTNAME144300"
	desc = "EVTDESC144300"
	picture = "goldrush"
	major = yes

	option = {
		name = "EVTOPTA144300"
		set_global_flag = fraser_gold_rush
		#Vancouver
		13 = {
			add_core = COL
			secede_province = THIS
			trade_goods = precious_metal
			life_rating = 20
			add_province_modifier = {
				name = "gold_rush"
				duration = 730
			}
		}

		# if England still controls Vancouver Island, Columbia is released.
		ENG = {
			release_vassal = COL
		}
	}
}
